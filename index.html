ⓘ Auto-connected to your Google Sheet. How to set it up →
Setup Steps:
1. Open Google Sheets → create a new sheet
2. Add headers in row 1: Date | Product Name | Brand | Price (MRP) | Quantity | Buy Price | Sell Price | Profit | Margin%
3. Go to Extensions → Apps Script
4. Paste the script below, save, then click Deploy → New deployment → Web App
5. Set "Who has access" to Anyone → Deploy → copy the URL
6. Click "✎ Change URL" above and paste the copied URL → Save

Apps Script Code: 
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);
    
    var buy  = parseFloat(data.buyPrice)  || 0;
    var sell = parseFloat(data.sellPrice) || 0;
    var qty  = parseFloat(data.quantity)  || 1;
    var profit = (sell - buy) * qty;
    var margin = buy > 0 ? ((sell - buy) / buy * 100).toFixed(1) + "%" : "0%";

    sheet.appendRow([
      new Date().toLocaleString('en-IN'),
      data.productName,
      data.brand,
      data.price !== '' ? parseFloat(data.price) : '',
      qty,
      buy,
      sell,
      profit,
      margin
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({status:"success"}))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:"error", message: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
